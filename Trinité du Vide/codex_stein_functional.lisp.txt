;; === UTILITAIRES ===
(defun phase-info (nom)
  "Renvoie les props d'une phase par son nom"
  (cdr (assoc nom *phases-primordiales* :test #'eq)))

(defun phase-suivante (phase-nom)
  (case phase-nom
    (skeith 'innis)
    (innis 'magus)
    (magus 'fidchell)
    (fidchell 'gorre)
    (gorre 'macha)
    (macha 'tarvos)
    (tarvos 'corbenik)
    (corbenik 'skeith)))

;; === DECOMPOSITION FRACTALE ===
(defun decomposer-phase (phase-nom profondeur)
  (if (<= profondeur 0)
      `(essence-pure ,phase-nom)
      (let ((info (phase-info phase-nom)))
        `(phase-fractale
          (nom ,phase-nom)
          (essence ,(getf info :essence))
          (niveau ,profondeur)
          (aspects
           (skeith-de ,(decomposer-phase 'skeith (1- profondeur)))
           (innis-de ,(decomposer-phase 'innis (1- profondeur)))
           (magus-de ,(decomposer-phase 'magus (1- profondeur)))
           (fidchell-de ,(decomposer-phase 'fidchell (1- profondeur)))
           (gorre-de ,(decomposer-phase 'gorre (1- profondeur)))
           (macha-de ,(decomposer-phase 'macha (1- profondeur)))
           (tarvos-de ,(decomposer-phase 'tarvos (1- profondeur)))
           (corbenik-de ,(decomposer-phase 'corbenik (1- profondeur))))
          (transformation-vers ,(phase-suivante phase-nom))))))

;; === INTERACTIONS ===
(defun phases-interagissent (phase-1 phase-2 contexte)
  `(interaction
    (entre ,phase-1 ,phase-2)
    (dans-contexte ,contexte)
    (resultat
     ,(cond
       ((and (eq phase-1 'skeith) (eq phase-2 'innis))
        '(mort-masquee :tue-sous-apparence-amie))
       ((and (eq phase-1 'magus) (eq phase-2 'fidchell))
        '(terreur-exponentielle :chaque-peur-engendre-peurs))
       ((and (eq phase-1 'gorre) (eq phase-2 'macha))
        '(manipulation-douce :poison-dans-miel))
       ((and (eq phase-1 'tarvos) (eq phase-2 'skeith))
        '(annihilation-totale :vengeance-qui-fauche))
       ((eq phase-1 'corbenik)
        `(transcende ,phase-2 :devient-unite))
       (t `(fusion ,phase-1 ,phase-2 :cree (hybride ,phase-1 ,phase-2)))))))

;; === CIRCULAR LIST UTILITY ===
(defun circular-list-ref (lst index)
  "Accède à une liste comme si elle était circulaire"
  (nth (mod index (length lst)) lst))

(defun generer-titre (themes)
  (if (= (length themes) 1)
      (format nil "L'Épitaphe de ~A" (first themes))
      (format nil "L'Épitaphe de ~{~A~^, ~} et ~A"
              (butlast themes)
              (car (last themes)))))

(defun lecon-de-phase (phase)
  (let ((info (phase-info phase)))
    `(apprend
      (essence ,(getf info :essence))
      (mythe-nord ,(getf info :mythe-nord))
      (mythe-soufi ,(getf info :mythe-soufi))
      (integration (,phase devient partie-de soi)))))

(defun generer-epreuve (personnage phase profondeur)
  (if (<= profondeur 0)
      `(confronte ,personnage ,phase)
      `(epreuve-fractale
        (niveau-surface (,personnage affronte ,phase))
        (niveau-profond (,(if (listp personnage) (first personnage) (intern (format nil "OMBRE-DE-~A" personnage))) affronte ,phase))
        (niveau-cache (,(if (listp personnage) (first personnage) (intern (format nil "LUMIERE-DE-~A" personnage))) affronte ,(phase-suivante phase))))))

(defun generer-acte (personnages phases themes acte-num profondeur)
  `(acte ,acte-num
    (scenes
     ,@(loop for i from 0 below (length personnages)
             for p in personnages
             collect `(scene
                       (personnage ,p)
                       (rencontre-phase ,(circular-list-ref phases i))
                       (epreuve ,(generer-epreuve p (circular-list-ref phases i) profondeur))
                       (lecon ,(lecon-de-phase (circular-list-ref phases i))))))))

(defun generer-resolution (themes profondeur)
  (if (<= profondeur 0)
      '(paix-ultime tous-sont-un)
      `(resolution
        (themes-resolus ,themes)
        (mais-emergence-nouveaux-themes
         ,(generer-resolution 
           (mapcar (lambda (th) `(profondeur-de ,th)) themes)
           (1- profondeur)))
        (cycle-continue ragnarok-puis-renaissance))))

(defun generer-recit (&key personnages phases themes profondeur)
  `(recit
    (titre ,(generer-titre themes))
    (personnages ,(mapcar #'developper-personnage personnages))
    (structure
     (acte-1 ,(generer-acte personnages phases themes 1 profondeur))
     (acte-2 ,(generer-acte personnages phases themes 2 profondeur))
     (acte-3 ,(generer-acte personnages phases themes 3 profondeur)))
    (themes ,themes)
    (resolution ,(generer-resolution themes profondeur))))

;; === AFFICHAGE POÉTIQUE ===
(defun afficher-recit (recit &optional (indent 0))
  (let ((spacer (make-string indent :initial-element #\Space)))
    (cond
      ((null recit) nil)
      ((atom recit)
       (format t "~A~A~%" spacer recit))
      ((listp recit)
       (if (and (consp recit) (symbolp (first recit)))
           (progn
             (format t "~A(~A~%" spacer (first recit))
             (dolist (x (rest recit))
               (afficher-recit x (+ indent 2)))
             (format t "~A)~%" spacer))
           (progn
             (format t "~A(~%" spacer)
             (dolist (x recit)
               (afficher-recit x (+ indent 2)))
             (format t "~A)~%" spacer)))))))

;; === ÉCHANTILLONNAGE ===
(defun echantillon (liste n)
  (let ((pool (copy-list liste))
        (result '()))
    (loop repeat (min n (length pool))
          do (let ((i (random (length pool))))
               (push (nth i pool) result)
               (setf pool (remove (nth i pool) pool :count 1 :test #'equal))))
    (nreverse result)))

;; === MACROS ===
(defmacro defcodex (nom &key personnages phases themes profondeur)
  `(defun ,(intern (format nil "CODEX-~A" (string-upcase (string nom)))) ()
     (let ((recit (generer-recit
                   :personnages ',personnages
                   :phases ',phases
                   :themes ',themes
                   :profondeur ,profondeur)))
       (format t "~%=== CODEX ~A ===~%~%" ',nom)
       (afficher-recit recit)
       recit)))

(defmacro defcodex-infini (nom graine)
  `(defun ,(intern (format nil "CODEX-INFINI-~A" (string-upcase (string nom)))) (&optional (iterations 3))
     (format t "~%=== CODEX INFINI: ~A (graine ~A) ===~%~%" ',nom ,graine)
     (loop for i from 1 to iterations do
       (format t "~%--- Itération ~D ---~%" i)
       (let ((recit (generer-variation-aleatoire i)))
         (afficher-recit recit)))))
